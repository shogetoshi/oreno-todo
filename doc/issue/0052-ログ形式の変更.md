- 出力するログフォーマットを変更したい
    - 下記のソースで出力されるものと同一にしたい
        - 出力する形式の部分以外は参考にする必要はないです。
    - 出力ファイルパスも下記プログラムに準拠させること

```
/**
 * タスクログ記録プラグイン
 * タスクの開始・停止・完了イベントをログファイルに記録します
 */

// ログファイルパスを保持する変数
let LOG_FILE_PATH = null

// 日時を JST フォーマットで取得
function getJSTTimeString() {
  const now = new Date()
  const jstOffset = 9 * 60 * 60 * 1000 // 9時間をミリ秒に変換
  const jstTime = new Date(now.getTime() + jstOffset)
  
  const year = jstTime.getUTCFullYear()
  const month = String(jstTime.getUTCMonth() + 1).padStart(2, '0')
  const day = String(jstTime.getUTCDate()).padStart(2, '0')
  const hours = String(jstTime.getUTCHours()).padStart(2, '0')
  const minutes = String(jstTime.getUTCMinutes()).padStart(2, '0')
  const seconds = String(jstTime.getUTCSeconds()).padStart(2, '0')
  
  return `${year}-${month}-${day} ${hours}:${minutes}:${seconds}`
}

// 今日の日付を YYYY_MM_DD 形式で取得
function getTodayDateString() {
  const now = new Date()
  const jstOffset = 9 * 60 * 60 * 1000 // 9時間をミリ秒に変換
  const jstTime = new Date(now.getTime() + jstOffset)
  
  const year = jstTime.getUTCFullYear()
  const month = String(jstTime.getUTCMonth() + 1).padStart(2, '0')
  const day = String(jstTime.getUTCDate()).padStart(2, '0')
  
  return `${year}_${month}_${day}`
}

// イベント時刻をフォーマット（HH:mm形式）
function formatEventTime(dateString) {
  if (!dateString) return ''
  const date = new Date(dateString)
  const jstOffset = 9 * 60 * 60 * 1000
  const jstTime = new Date(date.getTime() + jstOffset)
  const hours = String(jstTime.getUTCHours()).padStart(2, '0')
  const minutes = String(jstTime.getUTCMinutes()).padStart(2, '0')
  return `${hours}:${minutes}`
}

// ログエントリーを作成（Logseq形式）
function createLogEntry(data) {
  // Logseq形式: ## [#]([[id]]) content #taskcode
  let entry = '\n\n- ## '
  entry += `[#]([[${data.id}]]) `
  entry += data.content
  if (data.taskcode) {
    entry += ` #${data.taskcode}`
  }
  entry += '\n\t- \n'
  
  return entry
}

// ファイルに同一IDのエントリが存在するかチェック
async function isDuplicateEntry(context, filePath, id) {
  try {
    const result = await context.api.file.read(filePath)
    if (!result.success || !result.content) {
      // ファイルが存在しないか読み込めない場合は重複なし
      return false
    }
    
    // 行ごとにチェック（前方一致で確認）
    const lines = result.content.split('\n')
    const searchPattern = `- ## [#]([[${id}]]) `
    
    for (const line of lines) {
      if (line.startsWith(searchPattern)) {
        console.log(`既存のエントリを検出（ID: ${id}）、スキップします`)
        return true
      }
    }
    
    return false
  } catch (error) {
    console.error('ファイル読み込みエラー:', error)
    return false
  }
}

// プラグインのメインエクスポート
module.exports = {
  // プラグイン初期化
  init: function(context) {
    // contextからsettingsを取得
    const settings = context.settings
    
    // settingsから設定を読み込み（必須）
    if (!settings || !settings.config || !settings.config.logFilePath) {
      console.error('エラー: settings.jsonにlogFilePathが設定されていません')
      throw new Error('プラグイン設定エラー: settings.jsonのconfig.logFilePathを設定してください')
    }
    
    // {YYYY_MM_DD} プレースホルダーを今日の日付で置換
    LOG_FILE_PATH = settings.config.logFilePath.replace('{YYYY_MM_DD}', getTodayDateString())
    
    console.log('タスクログ記録プラグインを初期化しています...')
    console.log('設定ファイルから読み込んだログファイルパス:', LOG_FILE_PATH)
    
    // タスク開始イベント（通常タスクの処理開始時のみ記録）
    context.events.on('task:start', async (data) => {
      try {
        // Google Calendarイベントやクイックタスクは記録しない
        if (data.source === 'google-calendar' || data.isQuickTask) {
          return
        }
        
        // 重複チェック
        if (await isDuplicateEntry(context, LOG_FILE_PATH, data.id)) {
          return
        }
        
        const logEntry = createLogEntry(data)
        const result = await context.api.file.append(LOG_FILE_PATH, logEntry)
        
        if (!result.success) {
          console.error('ログ書き込みエラー:', result.error)
        } else {
          console.log('通常タスク開始をログに記録:', data.content)
        }
      } catch (error) {
        console.error('プラグインエラー (task:start):', error)
      }
    })
    
    // タスク完了イベント（Google Calendarイベントの完了時のみ記録）
    context.events.on('task:complete', async (data) => {
      try {
        // Google Calendarイベントの完了時のみ記録
        if (data.source !== 'google-calendar') {
          return
        }
        
        // 重複チェック
        if (await isDuplicateEntry(context, LOG_FILE_PATH, data.id)) {
          return
        }
        
        const logEntry = createLogEntry(data)
        const result = await context.api.file.append(LOG_FILE_PATH, logEntry)
        
        if (!result.success) {
          console.error('ログ書き込みエラー:', result.error)
        } else {
          console.log('Calendarイベント完了をログに記録:', data.content)
        }
      } catch (error) {
        console.error('プラグインエラー (task:complete):', error)
      }
    })
    
    // 初期化完了メッセージ
    
    console.log('タスクログ記録プラグインが有効になりました')
    console.log('ログファイル:', LOG_FILE_PATH)
  },
  
  // プラグイン終了時のクリーンアップ（オプション）
  destroy: function() {
    console.log('タスクログ記録プラグインを終了しています...')
  }
}
```
